Untuk test SSL:
docker compose run \
  --rm certbot certonly \
  --webroot --webroot-path /var/www/certbot \
  -d sindikat-pkm.com -d www.sindikat-pkm.com \
  --non-interactive \
  -m alfiangading81@gmail.com \
  --agree-tos --no-eff-email \
  --dry-run	

Untuk mendapatkan certificate (cuma menghilangkan --dry-run):
docker compose run \
  --rm certbot certonly \
  --webroot --webroot-path /var/www/certbot \
  -d sindikat-pkm.com -d www.sindikat-pkm.com \
  --non-interactive \
  -m alfiangading81@gmail.com \
  --agree-tos --no-eff-email


Untuk renew SSL certificate:
docker compose run --rm certbot renew
